include_directories(SYSTEM ${Boost_INCLUDE_DIRS})

macro(ADD_PAGMO_TESTCASE arg1)
    if(CMAKE_BUILD_TYPE STREQUAL "Debug")
        add_executable(${arg1} ${arg1}.cpp)
        target_link_libraries(${arg1} ${MANDATORY_LIBRARIES})
        add_test(${arg1} ${arg1})
    endif()
endmacro()
